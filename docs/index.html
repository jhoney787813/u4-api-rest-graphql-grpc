Introducción: GraphQL vs gRPC

        
Contexto, motivación y diferencias clave entre ambas tecnologías

      

    

    


      

        
1

        

          
Qué problema resuelven

          
Ambas tecnologías resuelven la comunicación entre servicios, pero con filosofías distintas. GraphQL expone un único endpoint flexible donde el cliente elige los datos; gRPC define contratos estrictos en un archivo .proto y genera código automáticamente.

        

      


      

        

          

            

              GraphQL — un endpoint flexible
            

            
# El CLIENTE decide qué campos recibir
query {
  productos {
    id
    nombre
    # precio no pedido = no viaja en la red
  }
}

# Endpoint único → POST /graphql
# Protocolo     → HTTP + JSON

          

        

        

          

            

              gRPC — contrato estricto en .proto
            

            
// El CONTRATO define todo de antemano
service ProductoService {
  rpc ListProductos(Empty)
      returns (ProductoList);
}

// Protocolo → HTTP/2 + binario
// Puerto    → 50051

Resolver de mutación con validación

          

            
app/resolvers/mutations.py — crear_producto

            
def crear_producto(db: Session, data: ProductoInput) -> ProductoType:
    if not data.nombre or not data.nombre.strip():
        raise ValidationError("El nombre no puede estar vacío")
    if data.precio < 0:
        raise ValidationError("El precio debe ser mayor o igual a 0")

    producto = Producto(
        nombre      = data.nombre.strip(),
        descripcion = data.descripcion,
        precio      = data.precio,
    )
    db.add(producto)
    db.commit()
    db.refresh(producto)
    return ProductoType(id=producto.id, nombre=producto.nombre,
                        descripcion=producto.descripcion, precio=producto.precio)

Excepciones tipadas con StatusCode gRPC

          

            
app/core/exceptions.py

            
import grpc

class GrpcError(Exception):
    def __init__(self, code: grpc.StatusCode, message: str):
        self.code    = code
        self.message = message

def handle_grpc_error(context, error: GrpcError):
    context.set_code(error.code)        # código HTTP-equivalente
    context.set_details(error.message)  # mensaje legible

def not_found_error(producto_id: int) -> GrpcError:
    return GrpcError(grpc.StatusCode.NOT_FOUND,
                     f"Producto con id {producto_id} no encontrado")

def invalid_argument_error(msg: str) -> GrpcError:
    return GrpcError(grpc.StatusCode.INVALID_ARGUMENT, msg)

def internal_error(msg: str) -> GrpcError:
    return GrpcError(grpc.StatusCode.INTERNAL, msg)

class ProductoServicer(producto_pb2_grpc.ProductoServiceServicer):

    def CreateProducto(self, request, context):
        db = SessionLocal()
        try:
            _validar_nombre(request.nombre)   # → INVALID_ARGUMENT si vacío
            _validar_precio(request.precio)   # → INVALID_ARGUMENT si < 0
            producto = Producto(
                nombre      = request.nombre.strip(),
                descripcion = request.descripcion or None,
                precio      = request.precio,
            )
            db.add(producto)
            db.commit()
            db.refresh(producto)
            return _modelo_a_response(producto)
        except GrpcError as e:
            handle_grpc_error(context, e)   # propaga StatusCode al cliente
            return producto_pb2.ProductoResponse()
        finally:
            db.close()                       # siempre cierra la sesión

UpdateProducto con flags de actualización parcial

          

            
app/services/producto_service.py — fragmento UpdateProducto

            
def UpdateProducto(self, request, context):
    db = SessionLocal()
    try:
        producto = db.query(Producto).filter(Producto.id == request.id).first()
        if not producto: raise not_found_error(request.id)

        # proto3: todos los campos tienen default → necesitamos flags explícitos
        if request.update_nombre:
            _validar_nombre(request.nombre)
            producto.nombre = request.nombre.strip()
        if request.update_descripcion:
            producto.descripcion = request.descripcion or None
        if request.update_precio:
            _validar_precio(request.precio)
            producto.precio = request.precio

        db.commit()
        db.refresh(producto)
        return _modelo_a_response(producto)
    except GrpcError as e:
        handle_grpc_error(context, e)
        return producto_pb2.ProductoResponse()
    finally:
        db.close()

          

          

            🔑
            En proto3 todos los campos tienen valor por defecto (0, "", false). Sin los flags update_*, sería imposible distinguir "actualizar precio a 0" de "no envié el precio".

import grpc
from app.grpc_client import producto_pb2, producto_pb2_grpc

GRPC_HOST = "localhost:50051"

def crear_producto_grpc(data: ProductoInput) -> ProductoType:
    channel = grpc.insecure_channel(GRPC_HOST)
    try:
        stub = producto_pb2_grpc.ProductoServiceStub(channel)
        req  = producto_pb2.ProductoRequest(
            nombre=data.nombre, descripcion=data.descripcion or "",
            precio=data.precio)
        resp = stub.CreateProducto(req)    # llamada gRPC
        return _respuesta_a_type(resp)
    except grpc.RpcError as e:
        raise Exception(f"Error gRPC [{e.code()}]: {e.details()}")
    finally:
        channel.close()

app/graphql_client/client.py

            
import json, urllib.request, urllib.error

GRAPHQL_URL = "http://localhost:8000/graphql"
_QUERY = """query { productos { id nombre descripcion precio } }"""

def listar_productos_graphql() -> List[dict]:
    payload = json.dumps({"query": _QUERY}).encode("utf-8")
    req = urllib.request.Request(
        GRAPHQL_URL, data=payload,
        headers={"Content-Type": "application/json"},
        method="POST")
    try:
        with urllib.request.urlopen(req, timeout=5) as resp:
            data = json.loads(resp.read().decode("utf-8"))
    except urllib.error.URLError as e:
        raise ConnectionError(f"No se pudo conectar a GraphQL: {e.reason}")
    return data["data"]["productos"]

Decisión de arquitectura — resumen visual

          

¿El cliente es un navegador o app móvil?
    Sí ──► GraphQL  (HTTP + JSON, GraphiQL, flexible)
    No ──► ¿Rendimiento o streaming crítico?
              Sí ──► gRPC    (Protobuf, bajo overhead, multi-lenguaje)
              No ──► GraphQL o REST (según complejidad del dominio)

¿Integración entre ambas tecnologías?
    GraphQL como cliente gRPC  ──► grpc_client/ + stubs pb2
    gRPC como cliente GraphQL  ──► graphql_client/ + urllib POST
          

        

      


      

        🎯
        Mensaje final del video: GraphQL y gRPC no son excluyentes. En arquitecturas reales, un API Gateway GraphQL puede orquestar múltiples microservicios gRPC internos, dando flexibilidad al cliente externo con alta eficiencia interna.
